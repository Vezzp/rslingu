\section{Синтаксический разбор предложений}\label{sec:SyntaxAnalysis}


Все команды данной группы имеют следующую сигнатуру:
\begin{tcolorbox}
    \rsCode{%
        \rsModifier[cmd]
        {}::
        \rsName[<syntax\_name>] \\\rsTab{}
        \rsOpt{
            \rsKwarg[color]:\rsType[bool]=false,
            \rsKwarg[phantom]:\rsType[bool]=false
        } \\\rsTab{}
        \rsReq{ \rsArg[текст:tl] }
    }
\end{tcolorbox}
где \rsName[<syntax\_name>] может быть одним из следующих значений:

\begin{table}[ht!]
    \centering
    \begin{tabular}{@{}llll@{}}
        \toprule

        \rsName[<syntax\_name>] & Значение       & \rsName[<syntax\_name>] & Значение \\\midrule

        \rsName[rsSubject]      & подлежащее
                                 &
        \rsName[rsPredicate]    & сказуемое
        \\\midrule

        \rsName[rsObject]       & дополнение
                                 &
        \rsName[rsAttribute]    & определение
        \\\midrule

        \rsName[rsAdverbial]    & обстоятельство
                                 &
        \\\midrule

        \bottomrule
    \end{tabular}
    \caption{Синтаксические~команды}
\end{table}

\subsection{Подлежащее}


\ExplSyntaxOn

\begingroup
\renewcommand{\arraystretch}{1.125}
\begin{table}[ht!]
    \centering
    \small
    \begin{tabular}{@{}ll@{}}
        \toprule

        \rsName*[rsSubject]
        \rsReq{ \rsArg[подлежащее] }
         &
        \rsSubject{подлежащее}

        \\\midrule

        \rsName*[rsSubject]
        \rsOpt{ \rsKwarg[color] }
        \rsReq{ \rsArg[подлежащее] }
         &
        \rsSubject[color]{подлежащее}

        \\\midrule

        \rsName*[rsSubject]
        \rsOpt{ \rsKwarg[color] }
        \rsReq{
            \rsName*[rsNoun] \rsReq{ \rsArg[подлежащее] }
            \rsSpace{}
            и
            \rsSpace{}
            \rsName*[rsNoun] \rsReq{ \rsArg[подлежащее] }
        }
         &
        \rsSubject[color]{\rsNoun{подлежащее}[им.п]~и~\rsNoun{подлежащее}[ср.р.]}

        \\\midrule

        \rsName*[rsSubject]
        \rsOpt{ \rsKwarg[color,phantom] }
        \rsReq{ \rsName*[rsNoun] \rsReq{ \rsArg[подлежащее] } }
         &
        \rsSubject[color,phantom]{\rsNoun{подлежащее}}

        \\\midrule

        \rsName*[rsSubject]
        \rsOpt{ \rsKwarg[color,phantom] }
        \rsReq{ \rsArg[подлежащее] }
         &
        \rsSubject[color,phantom]{подлежащее}

        \\\bottomrule
    \end{tabular}
    \caption{Использование~команды~подлежащего}
    \label{tab:subject-usage}
\end{table}
\endgroup
